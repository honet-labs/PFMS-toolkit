# Group: Linux
# Sub Group: Memory
module_begin
module_name Informations Memory Used
module_type generic_data_string
module_exec free -b | awk '/^Mem:/ {total=$2; used=$3; free=$4; cb=$6; printf "Percent Memory Used: %.2f%%\nCache + Buffer Used: %.2f%%\nFree Memory: %.2f%%\n", (used/total)*100, (cb/total)*100, (free/total)*100}'
module_end
